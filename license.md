## 🚀 Getting Started – Code Example

```cpp
#include <SDK.h>

void DrawESP() {
    // Get all players
    auto players = SDK::Game::GetPlayers();
    
    for (auto player : players) {
        // Get player position
        Vector position = player.GetPosition();
        
        // Draw ESP box
        SDK::Render::DrawBox(position, 32, 32, Color::Red());
        
        // Draw player name
        SDK::Render::DrawText(position.x, position.y - 20, 
                              player.GetName(), Color::White());
    }
}
```
